select CURRENCY, ((EXP(SUM(LOG( 1+PERFORMANCE_TWR)))-1)*100) as performance_totale_percent from POSITION_CASH_TWR_PERFORMANCE WHERE J_1 > to_date( '2019.07.01','YYYY:MM.DD')  and J_1 <= to_date( '2019.08.01','YYYY:MM.DD')
group by CURRENCY;
